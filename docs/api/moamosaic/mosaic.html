105def doMosaic(filelist, outfile, *, numthreads=DFLT_NUMTHREADS,
106        blocksize=DFLT_BLOCKSIZE, driver=DFLT_DRIVER, nullval=None,
107        creationoptions=None, outprojepsg=None,
108        outprojwktfile=None, outprojwkt=None, outXres=None,
109        outYres=None, resamplemethod=DFLT_RESAMPLEMETHOD):
110    """
111    From the given list of input raster files, create a single mosaic
112    output raster.
113
114    Parameters
115    ----------
116    filelist : List of str
117        List of filenames of input raster files
118    outfile : str
119        Name of output raster file
120    numthreads : int
121        Number of threads to use for reading input. These are in addition
122        to the main thread which manages everything else, including
123        writing the output
124    blocksize : int
125        Number of pixels in each square block of input (blocksize x blocksize)
126    driver : str
127        GDAL short name of format driver to use for output
128    creationoptions : List of str
129        List of 'NAME=VALUE' strings, giving GDAL creation options to go
130        with the selected format driver. If this is None, then some
131        sensible defaults are supplied for some known drivers.
132    nullval : int
133        Value to use as "no data" in input and output rasters. Default will
134        be taken from the input files, but this can be used to over-ride
135        that.
136    outprojepsg : int
137        EPSG number of projection for output file. Default projection
138        matches the input files
139    outprojwktfile : str
140        Name of text file containing WKT string for output projection
141    outprojwkt : str
142        WKT string for output projection
143    outXres, outYres : float
144        Desired output pxel size (X and Y directions). Default pixel
145        size matches the input files
146    resamplemethod : str
147        GDAL name of resampling method to use, if any resampling is required
148
149
150    Returns
151    -------
152    monitorInfo : Monitoring
153        An object of various bits of monitoring information, mainly
154        useful in development and testing. Most importantly, timing
155        information of various steps in the mosaicing process.
156
157    """
158    monitors = monitoring.Monitoring()
159    monitors.setParam('numthreads', numthreads)
160    monitors.setParam('blocksize', blocksize)
161    monitors.setParam('cpucount', cpu_count())
162
163    # Work out what we are going to do
164    monitors.setParam('numinfiles', len(filelist))
165    with monitors.timestamps.ctx("imginfodict"):
166        imgInfoDict = makeImgInfoDict(filelist)
167
168    with monitors.timestamps.ctx("projection"):
169        (filelist, tmpdir) = reproj.handleProjections(filelist,
170            imgInfoDict, outprojepsg, outprojwktfile, outprojwkt, outXres,
171            outYres, resamplemethod, nullval)
172
173    if nullval is None:
174        nullval = imgInfoDict[filelist[0]].nullVal
175
176    with monitors.timestamps.ctx("analysis"):
177        outImgInfo = makeOutputGrid(filelist, imgInfoDict, nullval)
178        blockList = makeOutputBlockList(outImgInfo, blocksize)
179
180        (blockListWithInputs, filesForBlock) = (
181            findInputsPerBlock(blockList, outImgInfo.transform, filelist,
182            imgInfoDict))
183        blockReadingList = makeBlockReadingList(blockListWithInputs)
184        blocksPerThread = divideBlocksByThread(blockReadingList, numthreads)
185
186    blockQ = queue.Queue()
187    poolClass = futures.ThreadPoolExecutor
188    numBands = imgInfoDict[filelist[0]].numBands
189
190    # Now do it all, using concurrent threads to read blocks into a queue
191    (outDs, overviewLevels) = openOutfile(outfile, driver, outImgInfo,
192        creationoptions)
193    statsAccumList = []
194    with monitors.timestamps.ctx("domosaic"):
195        for bandNum in range(1, numBands + 1):
196            statsAccum = StatsAccumulator(nullval)
197            statsAccumList.append(statsAccum)
198            with poolClass(max_workers=numthreads) as threadPool:
199                workerList = []
200                for i in range(numthreads):
201                    blocksToRead = blocksPerThread[i]
202                    worker = threadPool.submit(readFunc, blocksToRead, blockQ,
203                            bandNum, outImgInfo.nullVal)
204                    workerList.append(worker)
205
206                writeFunc(blockQ, outDs, outImgInfo, bandNum, blockList,
207                    filesForBlock, workerList, overviewLevels, statsAccum,
208                    monitors)
209
210    if tmpdir is not None:
211        shutil.rmtree(tmpdir)
212
213    return monitors

216def readFunc(blocksToRead, blockQ, bandNum, outNullVal):
217    """
218    This function is run by all the read workers, each with its own list
219    of blocks to read.
220
221    Parameters
222    ----------
223    blocksToRead : List of BlockSpecWithInputs objects
224    blockQ : queue.Queue
225        As each block is read, it is sent to the writer via this Queue
226    bandNum : int
227        GDAL band number of band to read (i.e. first band is 1)
228    outNullVal : int
229        Output null value. Used as the fill value for incomplete
230        blocks (i.e. when the output block falls partly off the edge of
231        the input raster)
232
233    """
234    blocksPerInfile = structures.BlocksByInfile()
235    for blockInfo in blocksToRead:
236        blocksPerInfile.blockToDo(blockInfo.filename, blockInfo.outblock)
237    gdalObjCache = structures.GdalObjCache()
238
239    i = 0
240    for blockInfo in blocksToRead:
241        filename = blockInfo.filename
242        (ds, band) = gdalObjCache.openBand(filename, bandNum)
243        inblock = blockInfo.inblock
244        (left, top, xsize, ysize) = (inblock.left, inblock.top,
245                inblock.xsize, inblock.ysize)
246        # Don't try to read outside the extent of the infile
247        left1 = max(left, 0)
248        top1 = max(top, 0)
249        right1 = min(left + xsize, ds.RasterXSize)
250        xsize1 = right1 - left1
251        bottom1 = min(top + ysize, ds.RasterYSize)
252        ysize1 = bottom1 - top1
253        arr = band.ReadAsArray(left1, top1, xsize1, ysize1)
254
255        # Now slot this possibly smaller array back into a full array,
256        # with null padding.
257        outArr = numpy.zeros((ysize, xsize), dtype=arr.dtype)
258        outArr.fill(outNullVal)
259        coloffset = max(0, -left)
260        rowoffset = max(0, -top)
261        outArr[rowoffset:(rowoffset + ysize1),
262               coloffset:(coloffset + xsize1)] = arr
263
264        # Put the full bloc into the blockQ, along with the associated
265        # block information
266        blockQ.put((blockInfo, outArr))
267
268        # If this input file is now done, we can close it.
269        blocksPerInfile.blockDone(filename, blockInfo.outblock)
270        if blocksPerInfile.countRemaining(filename) == 0:
271            gdalObjCache.closeBand(filename, bandNum)
272        i += 1

275def writeFunc(blockQ, outDs, outImgInfo, bandNum, blockList, filesForBlock,
276        workerList, overviewLevels, statsAccum, monitors):
277    """
278    Loop over all blocks of the output grid, and write them.
279
280    Input blocks are retrieved from the blockQ, and placed in a block
281    cache. When all inputs for a given output block are available,
282    that block is assembled for output, merging the inputs appropriately.
283    The resulting block is written to the output file.
284
285    The input blocks are deleted from the cache. All worker processes
286    are then checked for exceptions. Then move to the next output block.
287
288    This function runs continuously for a single band of the output file,
289    after which it returns. It will then be called again for the next band.
290
291    Parameters
292    ----------
293    blockQ : queue.Queue
294        Blocks of data are taken from this Queue
295    outDs : gdal.Dataset
296        Open Dataset of output file
297    outImgInfo : ImageInfo
298        All info about the output file
299    bandNum : int
300        GDAL band number (i.e. starts at 1) for band to write to outDs
301    blockList : List of BlockSpec
302        List of the blocks in the output grid. Blocks of data are written
303        in this order.
304    filesForBlock : dict
305        Key is a BlockSpec, value is a list of filenames which contribute
306        data to that block (i.e. intersect with it)
307    workerList : List of futures.Future
308        List of worker threads, so we can continually check them for
309        exceptions
310    overviewLevels: List of int
311        List of overview levels (as given to BuildOverviews()
312    statsAccum: StatsAccumulator
313        Object to manage incremental accumulators for single-pass statistics
314    monitors : Monitoring
315        A Monitoring object, mainly used to accumulate timing info
316
317    """
318    band = outDs.GetRasterBand(bandNum)
319    band.SetNoDataValue(outImgInfo.nullVal)
320
321    # Cache of blocks available to write
322    blockCache = structures.BlockCache()
323
324    numOutBlocks = len(blockList)
325    i = 0
326    while i < numOutBlocks:
327        # Get another block from the blockQ (if available), and cache it
328        if not blockQ.empty():
329            (blockInfo, arr) = blockQ.get_nowait()
330            filename = blockInfo.filename
331            outblock = blockInfo.outblock
332            blockCache.add(filename, outblock, arr)
333        else:
334            blockInfo = None
335            arr = None
336
337        outblock = blockList[i]
338        outArr = None
339
340        if outblock not in filesForBlock:
341            # This block does not intersect any input files, so
342            # just write nulls
343            numpyDtype = GDALTypeCodeToNumericTypeCode(outImgInfo.dataType)
344            outArr = numpy.zeros((outblock.ysize, outblock.xsize),
345                    dtype=numpyDtype)
346            outArr.fill(outImgInfo.nullVal)
347            band.WriteArray(outArr, outblock.left, outblock.top)
348            i += 1
349        elif blockInfo is not None or len(blockCache) > 0:
350            # If we actually got something from the blockQ, then we might
351            # be ready to write the current block
352
353            allInputBlocks = getInputsForBlock(blockCache, outblock,
354                    filesForBlock)
355            if allInputBlocks is not None:
356                outArr = mergeInputs(allInputBlocks, outImgInfo.nullVal)
357                band.WriteArray(outArr, outblock.left, outblock.top)
358
359                # Remove all inputs from cache
360                for filename in filesForBlock[outblock]:
361                    blockCache.remove(filename, outblock)
362
363                # Proceed to the next output block
364                i += 1
365
366        if outArr is not None:
367            # We actually wrote this block, so do pyramids and stats
368            writeBlockPyramids(band, outArr, overviewLevels, outblock.left,
369                outblock.top)
370            statsAccum.doStatsAccum(outArr)
371
372        checkReaderExceptions(workerList)
373
374        monitors.minMaxBlockCacheSize.update(len(blockCache))
375        monitors.minMaxBlockQueueSize.update(blockQ.qsize())
376
377    (minval, maxval, meanval, stddev, count) = statsAccum.finalStats()
378    if count > 0:
379        band.SetMetadataItem("STATISTICS_MINIMUM", str(minval))
380        band.SetMetadataItem("STATISTICS_MAXIMUM", str(maxval))
381        band.SetMetadataItem("STATISTICS_MEAN", str(meanval))
382        band.SetMetadataItem("STATISTICS_STDDEV", str(stddev))

385def checkReaderExceptions(workerList):
386    """
387    Check the read workers, in case one has raised an exception. The
388    elements of workerList are futures.Future objects. If a worker
389    has ended by raising an exception, then re-raise it.
390    """
391    for worker in workerList:
392        if worker.done():
393            e = worker.exception(timeout=0)
394            if e is not None:
395                raise e

418def makeOutputGrid(filelist, imgInfoDict, nullval):
419    """
420    Work out the extent of the whole mosaic. Return an ImageInfo
421    object of the output grid.
422    """
423    infoList = [imgInfoDict[fn] for fn in filelist]
424    boundsArray = numpy.array([(i.xMin, i.xMax, i.yMin, i.yMax)
425        for i in infoList])
426    xMin = boundsArray[:, 0].min()
427    xMax = boundsArray[:, 1].max()
428    yMin = boundsArray[:, 2].min()
429    yMax = boundsArray[:, 3].max()
430
431    firstImgInfo = imgInfoDict[filelist[0]]
432    outImgInfo = structures.ImageInfo(None)
433    outImgInfo.projection = firstImgInfo.projection
434    (xRes, yRes) = (firstImgInfo.xRes, firstImgInfo.yRes)
435    outImgInfo.ncols = int(round(((xMax - xMin) / xRes)))
436    outImgInfo.nrows = int(round(((yMax - yMin) / yRes)))
437    outImgInfo.transform = (xMin, xRes, 0.0, yMax, 0.0, -yRes)
438    outImgInfo.dataType = firstImgInfo.dataType
439    outImgInfo.numBands = firstImgInfo.numBands
440    outImgInfo.nullVal = firstImgInfo.nullVal
441    if nullval is not None:
442        outImgInfo.nullVal = nullval
443    return outImgInfo

446def makeOutputBlockList(outImgInfo, blocksize):
447    """
448    Given a pixel grid of the whole extent, divide it up into blocks.
449    Return a list of BlockSpec objects.
450    """
451    # Divide this up into blocks
452    (nrows, ncols) = (outImgInfo.nrows, outImgInfo.ncols)
453    blockList = []
454    top = 0
455    while top < nrows:
456        ysize = min(blocksize, (nrows - top))
457        # If what remains after this block is less than 25% of blocksize,
458        # then expand this block to the edge, to avoid very small blocks
459        if (nrows - (top + ysize)) < (blocksize // 4):
460            ysize = nrows - top
461
462        left = 0
463        while left < ncols:
464            xsize = min(blocksize, (ncols - left))
465            # Similarly avoid 'too narrow' blocks at the right-hand edge
466            if (ncols - (left + xsize)) < (blocksize // 4):
467                xsize = ncols - left
468
469            block = structures.BlockSpec(top, left, xsize, ysize)
470            blockList.append(block)
471            left += xsize
472
473        top += ysize
474    return blockList

488def findInputsPerBlock(blockList, outGeoTransform, filelist, imgInfoDict):
489    """
490    For every block, work out which input files intersect with it,
491    and the bounds of that block, in each file's pixel coordinate system.
492    """
493    blockListWithInputs = []
494    filesForBlock = {}
495    for block in blockList:
496        blockWithInputs = structures.BlockSpecWithInputs(block)
497
498        for filename in filelist:
499            imginfo = imgInfoDict[filename]
500            (fileLeft, fileTop, fileRight, fileBottom) = (
501                block.transformToFilePixelCoords(outGeoTransform, imginfo))
502            intersects = ((fileRight + 1) >= 0 and (fileBottom + 1) >= 0 and
503                fileLeft <= imginfo.ncols and fileTop <= imginfo.nrows)
504
505            if intersects:
506                xs = fileRight - fileLeft
507                ys = fileBottom - fileTop
508                inblock = structures.BlockSpec(fileTop, fileLeft, xs, ys)
509                blockWithInputs.add(filename, inblock)
510
511                if block not in filesForBlock:
512                    filesForBlock[block] = []
513                filesForBlock[block].append(filename)
514
515        if len(blockWithInputs.infilelist) > 0:
516            blockListWithInputs.append(blockWithInputs)
517
518    return (blockListWithInputs, filesForBlock)

551def getInputsForBlock(blockCache, outblock, filesForBlock):
552    """
553    Search the block cache for all the expected inputs for the current
554    block. If all are found, return a list of them. If any are still
555    missing from the cache, then just return None.
556    """
557    allInputsForBlock = []
558    i = 0
559    shp = None
560    missing = False
561    filelist = filesForBlock[outblock]
562    numFiles = len(filelist)
563    while i < numFiles and not missing:
564        filename = filelist[i]
565        k = blockCache.makeKey(filename, outblock)
566        if k in blockCache.cache:
567            (blockSpec, arr) = blockCache.cache[k]
568            # Check on array shape. They must all be the same shape
569            if shp is None:
570                shp = arr.shape
571            if arr.shape != shp:
572                msg = ("Block array mismatch at block {}\n".format(
573                       blockSpec) +
574                       "{}!={}\n{}".format(arr.shape, shp, filelist)
575                       )
576                raise ValueError(msg)
577
578            allInputsForBlock.append(arr)
579            i += 1
580        else:
581            missing = True
582            allInputsForBlock = None
583
584    return allInputsForBlock

587def openOutfile(outfile, driver, outImgInfo, creationoptions):
588    """
589    Open the output file.
590
591    Parameters
592    ----------
593    outfile : str
594    driver : str
595    outImgInfo : ImageInfo
596    creationoptions : List of str
597
598    Returns
599    -------
600    outDs : gdal.Dataset
601
602    """
603    if outfile is None:
604        raise ValueError("Must specify output file")
605
606    (nrows, ncols) = (outImgInfo.nrows, outImgInfo.ncols)
607    numBands = outImgInfo.numBands
608    datatype = outImgInfo.dataType
609    if creationoptions is None:
610        creationoptions = defaultCreationOptions[driver]
611    drvr = gdal.GetDriverByName(driver)
612    if drvr is None:
613        msg = "Driver {} not supported in this version of GDAL".format(driver)
614        raise ValueError(msg)
615
616    if os.path.exists(outfile):
617        drvr.Delete(outfile)
618    ds = drvr.Create(outfile, ncols, nrows, numBands, datatype,
619        creationoptions)
620    ds.SetGeoTransform(outImgInfo.transform)
621    ds.SetProjection(outImgInfo.projection)
622
623    # Work out a list of overview levels, starting with 4, until the raster
624    # size (in largest direction) is smaller then finalOutSize.
625    outSize = max(ds.RasterXSize, ds.RasterYSize)
626    finalOutSize = 1024
627    overviewLevels = []
628    i = 2
629    while ((outSize // (2 ** i)) >= finalOutSize):
630        overviewLevels.append(2 ** i)
631        i += 1
632
633    # Create the empty pyramid layers on the dataset. Currently only
634    # support NEAREST
635    aggType = "NEAREST"
636    ds.BuildOverviews(aggType, overviewLevels)
637
638    return (ds, overviewLevels)

641def mergeInputs(allInputsForBlock, outNullVal):

698def writeBlockPyramids(band, arr, overviewLevels, xOff, yOff):

701    given as arr. Uses nearest neighbour sampling to sub-sample the array.

713        # The xOff/yOff of the block within the sub-sampled raster

715        yOff_sub = yOff // lvl
716        # The actual number of rows and cols to write, ensuring we
717        # do not go off the edges

719        nr = band_ov.YSize - yOff_sub

725    """
726    Accumulator for statistics for a single band.
727    """
728    def __init__(self, nullval):
729        self.nullval = nullval
730        self.minval = None
731        self.maxval = None
732        self.sum = 0
733        self.ssq = 0
734        self.count = 0
735
736    def doStatsAccum(self, arr):
737        """
738        Accumulate basic stats for the given array
739        """
740        if self.nullval is None:
741            values = arr.flatten()
742        elif numpy.isnan(self.nullval):
743            values = arr[~numpy.isnan(arr)]
744        else:
745            values = arr[arr != self.nullval]
746        if len(values) > 0:
747            self.sum += values.astype(numpy.float64).sum()
748            self.ssq += (values.astype(numpy.float64)**2).sum()
749            self.count += values.size
750            minval = values.min()
751            if self.minval is None or minval < self.minval:
752                self.minval = minval
753            maxval = values.max()
754            if self.maxval is None or maxval > self.maxval:
755                self.maxval = maxval
756
757    def finalStats(self):
758        """
759        Return the final values of the four basic statistics
760        (minval, maxval, mean, stddev)
761        """
762        meanval = None
763        stddev = None
764        if self.count > 0:
765            meanval = self.sum / self.count
766            variance = self.ssq / self.count - meanval ** 2
767            stddev = 0.0
768            # In case some rounding error made variance negative
769            if variance >= 0:
770                stddev = numpy.sqrt(variance)
771
772        return (self.minval, self.maxval, meanval, stddev, self.count)
