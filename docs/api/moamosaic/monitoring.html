 64class TimeStampSet():
 65    """
 66    A class for managing lots of time stamps at different points within
 67    a program. This has grown and changed, but I feel like there is
 68    something more generally useful in this.
 69    """
 70    def __init__(self, keySep=':'):
 71        self.stamps = {}
 72        self.keySep = keySep
 73
 74    @contextlib.contextmanager
 75    def ctx(self, name):
 76        """
 77        Allow use of with statements for start/end timing
 78        """
 79        startTime = time.time()
 80        yield
 81        endTime = time.time()
 82        startKey = self.__makekey(name, TS_START)
 83        self.stamps[startKey] = startTime
 84        endKey = self.__makekey(name, TS_END)
 85        self.stamps[endKey] = endTime
 86
 87    def __makekey(self, name, startEnd):
 88        if self.keySep in name:
 89            raise ValueError("Time stamp name cannot include '{}'".format(
 90                self.keySep))
 91        return "{}{}{}".format(name, self.keySep, startEnd)
 92
 93    def stamp(self, name, startEnd):
 94        key = self.__makekey(name, startEnd)
 95
 96        self.stamps[key] = time.time()
 97
 98    def getStamp(self, name, startEnd):
 99        """
100        Return the stamp for the given name/startEnd
101        """
102        key = self.__makekey(name, startEnd)
103        return self.stamps[key]
104
105    def namesByPrefix(self, prefix):
106        """
107        Return a list of all names beginning with given prefix
108        """
109        nameList = list(set([key.split(self.keySep)[0]
110                for key in self.stamps]))
111        nameList = [name for name in nameList if name.startswith(prefix)]
112        return nameList
113
114    def timeSpentByPrefix(self, prefix):
115        """
116        Add up the total time spent, for the given
117        prefix
118        """
119        tot = 0
120        nameList = self.namesByPrefix(prefix)
121        for name in nameList:
122            startTime = self.getStamp(name, TS_START)
123            endTime = self.getStamp(name, TS_END)
124            diff = endTime - startTime
125            tot += diff
126        return round(tot, 2)
127
128    def timeElapsedByPrefix(self, prefix):
129        """
130        Find the elapsed wall time spent doing anything matching
131        prefix. This is the time difference between the earliest
132        and latest time stamps.
133        """
134        nameList = self.namesByPrefix(prefix)
135        stampList = []
136        for name in nameList:
137            stamp = self.getStamp(name, TS_START)
138            stampList.append((stamp, TS_START))
139            stamp = self.getStamp(name, TS_END)
140            stampList.append((stamp, TS_END))
141        # Sort the events into chronological order
142        stampList = sorted(stampList)
143        # Count when we were in at least one action
144        tot = 0
145        count = 0
146        prevStamp = stampList[0][0]
147        for (stamp, flag) in stampList:
148            if count > 0:
149                # There is at least one 'prefix' action still going, so
150                # this time period counts, add it to the total
151                tot += (stamp - prevStamp)
152            prevStamp = stamp
153
154            if flag == TS_START:
155                count += 1
156            else:
157                count -= 1
158        return round(tot, 2)
159
160    def avgTimeByPrefix(self, prefix):
161        """
162        Over all start/end pairs matching the given prefix, find the
163        average time taken from start to end.
164        """
165        nameList = self.namesByPrefix(prefix)
166        durationList = []
167        for name in nameList:
168            startTime = self.getStamp(name, TS_START)
169            endTime = self.getStamp(name, TS_END)
170            duration = endTime - startTime
171            durationList.append(duration)
172        avgTime = sum(durationList) / len(durationList)
173        return round(avgTime, 3)
174
175    def pcntOverlapByGroup(self, groupList):
176        """
177        Percentage overlapping calculations within each group of
178        the list. Return array of percentages, one for each group.
179        """
180        pcntList = []
181        for group in groupList:
182            stampList = []
183            for name in group:
184                startStamp = self.getStamp(name, TS_START)
185                stampList.append((startStamp, TS_START))
186                endStamp = self.getStamp(name, TS_END)
187                stampList.append((endStamp, TS_END))
188            # Sort the events into chronological order
189            stampList = sorted(stampList)
190
191            # Count when we were in more than one action, i.e. there was some
192            # overlap going on.
193            totInOverlap = 0
194            count = 0
195            prevStamp = stampList[0][0]
196            for (stamp, flag) in stampList:
197                if count > 1:
198                    # There is more than one action still going, so
199                    # this time period counts, add it to the total
200                    totInOverlap += (stamp - prevStamp)
201                prevStamp = stamp
202
203                if flag == TS_START:
204                    count += 1
205                else:
206                    count -= 1
207
208            totElapsed = stampList[-1][0] - stampList[0][0]
209            pcnt = 100 * totInOverlap / totElapsed
210            pcntList.append(pcnt)
211        return numpy.array(pcntList)
212
213    def merge(self, other):
214        """
215        Merge another timestamps instance into this one
216        """
217        self.stamps.update(other.stamps)
